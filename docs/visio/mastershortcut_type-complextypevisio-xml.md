```XML
          <xs:complexType name="MasterShortcut_Type">
          
          <xs:all>
    <xs:element name="Icon"  type="Icon_Type"

     maxOccurs="1"

    <xs:attribute name="ID"
  type="xsd:unsignedInt"
     use="required"

  type="xsd:string"

    />
    <xs:attribute name="IsCustomNameU"
  type="xsd:boolean"
    />
    <xs:attribute name="IconSize"
  type="xsd:unsignedShort"
    />
```
